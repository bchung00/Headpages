CREATE VIEW group_popularity AS
    SELECT G.name, COUNT(*)
    FROM groups G
    JOIN group_contains_user GCU ON G.GID = GCU.GID
    GROUP BY G.GID
    ORDER BY COUNT(*) DESC;
